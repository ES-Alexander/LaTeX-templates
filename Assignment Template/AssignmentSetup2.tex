% Command for entering questions
\makeatletter
\newcommand{\que}[2][\@nil]{
  \def\tmp{#1} % define a temporary variable for checking purposes
  
  \color{black!60} % set the question text to grey
  % conditional code depending on existence of optional argument
  \ifx\tmp\@nnil
    \item
  \else
    \item[#1]
  \fi
  
  \textbf{#2} % bold (and grey) question text
  \color{black} % return the following text to black
}
\makeatother
