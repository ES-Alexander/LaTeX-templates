				&= \left| \begin{pmatrix} #2 & #3\\ #4 & #5 \end{pmatrix} -

				&= \begin{vmatrix}
						#2 - \lambda & #3\\
						#4 & #5 - \lambda
					\end{vmatrix}\\
				&= (#2-\lambda)(#5 - \lambda) - (#3)(#4)}

					\end{pmatrix} \begin{pmatrix}
						#3 \\ #4
					\end{pmatrix} &= \begin{pmatrix}
						0 \\ 0
					\end{pmatrix}}
