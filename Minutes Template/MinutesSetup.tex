% an 'owner' command for specifying who owns a given topic
\newcommand{\owner}[1]{\textit{(#1)}}

% a 'topic' command for bolded topics in the agenda and minutes
\makeatletter
\newcommand{\topic}[2][\@nil]{
  \def\tmp{#1}
  
  \item \textbf{#2}
  \ifx\tmp\@nnil
    % do nothing
  \else
    \owner{#1}
  \fi
}
\makeatother
