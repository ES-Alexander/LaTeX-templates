% use as \ingredients{\ing{...} \ing{...}}
\newcommand{\ingredients}[1]{
  \textbf{Ingredients}
  \begin{itemize}[label=-] 
    #1 
  \end{itemize}
}

% use as \ing[notes]{amount}{units}{ingredient}
\newcommand{\ing}[4][]{\item #2#3 #4
  \ifx&#1&%
  \else
    (#1)
  \fi
}
  
% use as \steps{\step{...} \step{...}}
\newcommand{\steps}[1]{
  \textbf{Steps}
  \begin{enumerate} 
    #1 
  \end{enumerate}
}

% use as \ing[note]{instruction}, or as \ing[note step]{}
\newcommand{\step}[2][]{
  \ifx&#2&%
    \item[*] #1
  \else
	  \item #2
	  \ifx&#1&%
	  \else
	    (#1)
	  \fi
  \fi
}
