    \recipe{Pizza}{
      \ing[tastes better with plain]{500}{g}{wholemeal flour}
      \ing{14}{g}{yeast}
    	\ing{10}{g}{salt}
    	\ing{300}{g}{water}
    	\ing{50}{g}{olive oil}
    	\ing[for extra crunch]{}{}{polenta for rolling}
    }{
      \step[sprinkle with fingers]{Mix the dry ingredients in a bowl}
      \step{Add the water and oil}
      \step{Mix together until individual ingredients aren't loose}
      \step{Knead for 5-10 mins by pinching the sides of the dough, pulling
            it up, and flipping it over itself and onto the bench}
      \step[ideally until doubled in size]{Let rise for $>$30 mins or
            overnight in the fridge}
      \step{Weigh the dough into thirds, and roll out on polenta until the
                desired thickness}
      \step{Fold over the edges and press them down into the dough to form 
            a crust}
      \step[$250^\text{o}$C fan-forced $\rightarrow$ about 7-10 mins for a
            single pizza, more pizzas at once slows it down a bit]{
            Add desired toppings (e.g. passata, cheese, fresh basil) and 
            bake in hot oven (hotter $\rightarrow$ faster) until crust
            begins to brown}
      \step[Dough will last up to about a week in the fridge before it needs
            to be used or refreshed (add more flour and water)]{}
    }
